# Решить задачу (двумерный массив)


## Описание

**Двумерный массив**

Ваша задача - изменить числа на главной диагонали. Все отрицательные нужно заменить на 0, а положительные на 1.

Пример:

Входной массив

[

  [-1,  4, -5, -9,  3 ],

  [ 6, -4, -7,  4, -5 ],

  [ 3,  5,  0, -9, -1 ],

  [ 1,  5, -7, -8, -9 ],

  [-3,  2,  1, -5,  6 ]

]

Выходной массив

[

  [ 0,  4, -5, -9,  3 ],

  [ 6,  0, -7,  4, -5 ],

  [ 3,  5,  1, -9, -1 ],

  [ 1,  5, -7,  0, -9 ],

  [-3,  2,  1, -5,  1 ]

]
